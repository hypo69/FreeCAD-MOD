def normalize_answer(text: str) -> str:
    """–§—É–Ω–∫—Ü–∏—è –æ—á–∏—â–∞–µ—Ç –æ—Ç–≤–µ—Ç –æ—Ç Markdown-–æ–±—ë—Ä—Ç–æ–∫ –≤—Ä–æ–¥–µ ```text ... ```."""
    text = text.strip()
    if text.startswith("```") and text.endswith("```"):
        lines = text.split("\n")
        if len(lines) >= 2:
            return "\n".join(lines[1:-1])
    if text.startswith("text\n"):
        text = text[5:]
    return text.strip()

def get_image_bytes(image_path: Path) -> Optional[bytes]:
    """–§—É–Ω–∫—Ü–∏—è —á–∏—Ç–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –∫–∞–∫ –±–∞–π—Ç—ã."""
    try:
        with open(image_path, "rb") as f:
            return f.read()
    except Exception as ex:
        log_error(f"–ù–µ —É–¥–∞–ª–æ—Å—å –ø—Ä–æ—á–∏—Ç–∞—Ç—å –±–∞–π—Ç—ã –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏–∑ —Ñ–∞–π–ª–∞: {image_path}")
        return None

        """
        self.api_key = api_key

    def ask(self, q: str, attempts: int = 15, save_dialogue: bool = False,
            clean_response: bool = True, context: Optional[Union[str, List[str]]] = None) -> Optional[str]:
        """
        –ú–µ—Ç–æ–¥ —Å–∏–Ω—Ö—Ä–æ–Ω–Ω–æ –æ—Ç–ø—Ä–∞–≤–ª—è–µ—Ç —Ç–µ–∫—Å—Ç–æ–≤—ã–π –∑–∞–ø—Ä–æ—Å –º–æ–¥–µ–ª–∏ –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ—Ç–≤–µ—Ç.

        Args:
            q (str): –¢–µ–∫—Å—Ç–æ–≤—ã–π –∑–∞–ø—Ä–æ—Å –∫ –º–æ–¥–µ–ª–∏.
            attempts (int): –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–æ–ø—ã—Ç–æ–∫ –æ—Ç–ø—Ä–∞–≤–∫–∏ –∑–∞–ø—Ä–æ—Å–∞.
            save_dialogue (bool): –§–ª–∞–≥ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è –¥–∏–∞–ª–æ–≥–∞ (–Ω–µ —Ä–µ–∞–ª–∏–∑–æ–≤–∞–Ω –≤ FreeCAD).
            clean_response (bool): –§–ª–∞–≥ –æ—á–∏—Å—Ç–∫–∏ –æ—Ç–≤–µ—Ç–∞ –æ—Ç —Ä–∞–∑–º–µ—Ç–∫–∏ –∫–æ–¥–∞.
            context (Optional[Union[str, List[str]]]): –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–π –∫–æ–Ω—Ç–µ–∫—Å—Ç –¥–ª—è –º–æ–¥–µ–ª–∏.

        Returns:
            Optional[str]: –¢–µ–∫—Å—Ç–æ–≤—ã–π –æ—Ç–≤–µ—Ç –º–æ–¥–µ–ª–∏ –∏–ª–∏ `None` –≤ —Å–ª—É—á–∞–µ –Ω–µ—É–¥–∞—á–∏.
        """
        content_to_send: List[Any] = []
        if context:
            context_str = "\n".join(context) if isinstance(context, list) else context
            content_to_send.append(f"–ö–æ–Ω—Ç–µ–∫—Å—Ç:\n{context_str}\n")
            log_debug(f"–ö–æ–Ω—Ç–µ–∫—Å—Ç RAG –¥–æ–±–∞–≤–ª–µ–Ω –≤ –∑–∞–ø—Ä–æ—Å (–¥–ª–∏–Ω–∞: {len(context_str)} —Å–∏–º–≤–æ–ª–æ–≤).")
        content_to_send.append(q)

        for attempt in range(attempts):
            try:
                response = self.model.generate_content(content_to_send)
                if hasattr(response, 'text') and response.text:
                    response_text = response.text
                    return normalize_answer(response_text) if clean_response else response_text
                else:
                    sleep_time = INITIAL_RETRY_SLEEP_SECONDS ** attempt
                    log_debug(f"–û—Ç –º–æ–¥–µ–ª–∏ –Ω–µ –ø–æ–ª—É—á–µ–Ω –æ—Ç–≤–µ—Ç. –ü–æ–ø—ã—Ç–∫–∞: {attempt + 1}/{attempts}. –ü–∞—É–∑–∞: {sleep_time} —Å–µ–∫.")
                    time.sleep(sleep_time)
                    continue
            except ResourceExhausted:
                log_error("–ò—Å—á–µ—Ä–ø–∞–Ω –ª–∏–º–∏—Ç –∫–≤–æ—Ç—ã.")
                return "ResourceExhausted"
            except (DefaultCredentialsError, RefreshError):
                log_error("–û—à–∏–±–∫–∞ –∞—É—Ç–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ü–∏–∏.")
                return None
            except (GatewayTimeout, ServiceUnavailable):
                if attempt >= SERVICE_UNAVAILABLE_MAX_ATTEMPTS:
                    log_error(f"–°–µ—Ä–≤–∏—Å –Ω–µ–¥–æ—Å—Ç—É–ø–µ–Ω –ø–æ—Å–ª–µ {SERVICE_UNAVAILABLE_MAX_ATTEMPTS} –ø–æ–ø—ã—Ç–æ–∫.")
                    break
                sleep_time_service = INITIAL_RETRY_SLEEP_SECONDS**attempt + SERVICE_RETRY_SLEEP_SECONDS_BASE
                log_error(f"–°–µ—Ä–≤–∏—Å –Ω–µ–¥–æ—Å—Ç—É–ø–µ–Ω. –ü–æ–ø—ã—Ç–∫–∞: {attempt + 1}/{attempts}. –ü–∞—É–∑–∞: {sleep_time_service} —Å–µ–∫.")
                time.sleep(sleep_time_service)
                continue
            except (ValueError, TypeError):
                if attempt >= INVALID_INPUT_MAX_ATTEMPTS:
                    log_error(f"–û—à–∏–±–∫–∞ –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö –ø–æ—Å–ª–µ {INVALID_INPUT_MAX_ATTEMPTS} –ø–æ–ø—ã—Ç–æ–∫.")
                    break
                time.sleep(5)
                continue
            except Exception as ex:
                log_error("–ù–µ–æ–∂–∏–¥–∞–Ω–Ω–∞—è –æ—à–∏–±–∫–∞.")
                return None

        log_error(f"–ù–µ —É–¥–∞–ª–æ—Å—å –ø–æ–ª—É—á–∏—Ç—å –æ—Ç–≤–µ—Ç –æ—Ç –º–æ–¥–µ–ª–∏ –ø–æ—Å–ª–µ {attempts} –ø–æ–ø—ã—Ç–æ–∫.")
        return None

    def describe_image(
        self, image: Path | bytes, mime_type: Optional[str] = 'image/jpeg', prompt: Optional[str] = ''
    ) -> Optional[str]:
        """
        –§—É–Ω–∫—Ü–∏—è –æ—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –≤ –º–æ–¥–µ–ª—å Gemini –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –µ–≥–æ –æ–ø–∏—Å–∞–Ω–∏–µ.

        Args:
            image (Path | bytes): –ü—É—Ç—å –∫ —Ñ–∞–π–ª—É –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏–ª–∏ –±–∞–π—Ç—ã.
            mime_type (Optional[str]): MIME-—Ç–∏–ø –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è.
            prompt (Optional[str]): –¢–µ–∫—Å—Ç–æ–≤—ã–π –ø—Ä–æ–º–ø—Ç –¥–ª—è –º–æ–¥–µ–ª–∏.

        Returns:
            Optional[str]: –¢–µ–∫—Å—Ç–æ–≤–æ–µ –æ–ø–∏—Å–∞–Ω–∏–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏–ª–∏ `None` –ø—Ä–∏ –æ—à–∏–±–∫–µ.
        """
        image_data: bytes
        if isinstance(image, Path):
            img_bytes = get_image_bytes(image)
            if img_bytes is None:
                return None
            image_data = img_bytes
        elif isinstance(image, bytes):
            image_data = image
        else:
            log_error(f"–ù–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–π —Ç–∏–ø –¥–ª—è 'image'. –û–∂–∏–¥–∞–µ—Ç—Å—è Path –∏–ª–∏ bytes, –ø–æ–ª—É—á–µ–Ω–æ: {type(image)}")
            return None

        content_parts: List[Any] = []
        if prompt:
            content_parts.append(prompt)
        content_parts.append(image_data)

        try:
            response = self.model.generate_content(content_parts)
            if hasattr(response, 'text') and response.text:
                return response.text
            else:
                log_info(f"–ú–æ–¥–µ–ª—å –≤–µ—Ä–Ω—É–ª–∞ –æ—Ç–≤–µ—Ç –±–µ–∑ —Ç–µ–∫—Å—Ç–∞: {response}")
                return None
        except Exception as ex:
            log_error("–ü—Ä–æ–∏–∑–æ—à–ª–∞ –æ—à–∏–±–∫–∞ –ø—Ä–∏ –æ–±—Ä–∞–±–æ—Ç–∫–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è.")
            return None
